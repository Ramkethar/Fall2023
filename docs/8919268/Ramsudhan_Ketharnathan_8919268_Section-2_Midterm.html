import pandas as pd

# TASK 1: Loading dataset
data = pd.read_csv("C:/Users/91902/Downloads/housing_dataset.csv")

# TASK 2: Calculating IQR for 'median_house_value'
Q1 = data['median_house_value'].quantile(0.25)

Q3 = data['median_house_value'].quantile(0.75)

IQR = Q3 - Q1

# TASK 3: Define lower and upper bounds for outliers
lwr_bound = Q1 - 1.5 * IQR

upr_bound = Q3 + 1.5 * IQR

# TASK 4: Detecting outliers
outliers = data[(data['median_house_value'] < lwr_bound) | (data['median_house_value'] > upr_bound)]

print("Outliers- whiskers approach:")

#output
print(outliers)

# TASK 1: Checking for missing values in numerical features

numerical_features = ['housing_median_age', 'total_rooms', 'total_bedrooms', 'population', 'households', 'median_income', 'median_house_value']

for feature in numerical_features:


    if data[feature].isnull().any():
        
        # Using median for skewed data 
        median = data[feature].median()
        
        data[feature].fillna(median, inplace=True)




# TASK 2: Checking for missing values in categorical feature which is 'ocean_proximity'

if data[categorical_feature].isnull().any():

    # Imputing with mode
